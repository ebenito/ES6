for of

    
        Itera en objetos iterables, tipo arrays.

        let apellidos = [ 'pérez', 'garcía', 'gómez'];
        for (let setname of apellidos) {
            console.log(setname);
        }

for in

    
        Itera sobre todas las propiedades enumerables
        del objeto, dando el valor de la clave
        de objeto a nuestra variable.

        let apellidos = ['perez', 'garcia', 'gomez'];
        for (let index in apellidos) {
            console.log('apellidos[' + index + '] = ' + apellidos[index]);
        }
